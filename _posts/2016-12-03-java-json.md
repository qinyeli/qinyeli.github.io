A quick note on how to generate JSON in Java.

```java
import java.io.FileWriter;
import java.io.IOException;
import org.json.JSONObject;
import org.json.JSONArray;

...

// A JSONObject is an unordered collection of name/value pairs. Add a few name/value pairs.
JSONObject object = new JSONObject();

// Key must be a string
// Value can be anything from Integer, Long, String, to ArrayList, etc.
object.put("first_name", Harry);
object.put("last_name", Potter);


// JSONArray is an array of JSONObjects
JSONArray array = new JSONArray();
array.put(object);

try {
    FileWriter file = new FileWriter(System.getProperty("user.dir")+"/output.json");
    file.write(users_info.toString());
    file.flush();
    file.close();

} catch (IOException e) {
    e.printStackTrace();
}
```
